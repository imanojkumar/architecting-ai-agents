% ----- Fonts ------------------------------------------------
\usepackage{fontenc}        % T1 font encoding
\usepackage{lmodern}        % Latin Modern — scalable CM replacement
\usepackage{microtype}      % microtypography (better justification)

% ----- Colour -----------------------------------------------
\usepackage{xcolor}
\definecolor{navy}{HTML}{1B2A4A}
\definecolor{teal}{HTML}{0D7680}
\definecolor{gold}{HTML}{C8972B}

% ----- Hyperlinks -------------------------------------------
% (Quarto sets hyperref; we only adjust colours here)
\hypersetup{
  colorlinks = true,
  linkcolor  = navy,
  urlcolor   = teal,
  citecolor  = teal
}

% ----- Chapter / Part styling (scrbook) --------------------
\usepackage{titlesec}

% Part title
\titleformat{\part}[display]
  {\normalfont\Huge\bfseries\color{navy}}
  {\partname~\thepart}{20pt}{\Huge}

% Chapter title
\titleformat{\chapter}[display]
  {\normalfont\huge\bfseries\color{navy}}
  {\chaptertitlename~\thechapter}{16pt}{\Huge}

% Section
\titleformat{\section}
  {\normalfont\Large\bfseries\color{navy}}
  {\thesection}{1em}{}

% Subsection
\titleformat{\subsection}
  {\normalfont\large\bfseries\color{teal}}
  {\thesubsection}{1em}{}

% ----- Code listings ----------------------------------------
\usepackage{listings}
\lstset{
  basicstyle   = \ttfamily\small,
  keywordstyle = \color{teal}\bfseries,

  stringstyle  = \color{gold},
  breaklines   = true,
  frame        = leftline,
  framerule    = 2pt,
  rulecolor    = \color{teal},
  showstringspaces = false
}
